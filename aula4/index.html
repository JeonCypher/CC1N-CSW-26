    # Calculando o discriminante (delta)
    delta = b**2 - 4*a*c

    if delta < 0:
        print("A equação não possui raízes reais.")
    elif delta == 0:
        x = -b / (2*a)
        print("A equação possui uma raiz real:")
        print("x =", x)
    else:
        x1 = (-b + math.sqrt(delta)) / (2*a)
        x2 = (-b - math.sqrt(delta)) / (2*a)
        print("A equação possui duas raízes reais:")
        print("x1 =", x1)
        print("x2 =", x2)
